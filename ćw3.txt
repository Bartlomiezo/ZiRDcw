1.
SELECT last_name, salary,
       DENSE_RANK() OVER (ORDER BY salary DESC) AS ranking_plac
FROM employees;

2.
SELECT last_name, salary,
       SUM(salary) OVER () AS suma_wszystkich_plac
FROM employees;

3.
CREATE TABLE sales AS SELECT * FROM HR.sales;
CREATE TABLE products AS SELECT * FROM HR.products;

4.
SELECT e.last_name, p.product_name,
       SUM(s.price * s.quantity) OVER (PARTITION BY e.employee_id ORDER BY s.sale_date) AS skumulowana_wartosc,
       RANK() OVER (ORDER BY s.price * s.quantity DESC) AS ranking_wartosci
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

5.
SELECT e.last_name, p.product_name, s.price,
       COUNT(*) OVER (PARTITION BY p.product_id, s.sale_date) AS liczba_transakcji_dzis,
       SUM(s.price * s.quantity) OVER (PARTITION BY p.product_id, s.sale_date) AS suma_dzienna,
       LAG(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS poprzednia_cena,
       LEAD(s.price) OVER (PARTITION BY p.product_id ORDER BY s.sale_date) AS kolejna_cena
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

6.
SELECT p.product_name, s.price,
       SUM(s.price * s.quantity) OVER (PARTITION BY p.product_id, TRUNC(s.sale_date, 'MM')) AS suma_miesieczna_calosc,
       SUM(s.price * s.quantity) OVER (PARTITION BY p.product_id, TRUNC(s.sale_date, 'MM') ORDER BY s.sale_date) AS suma_miesieczna_rosnaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

7.
SELECT p1.product_name, p1.product_category,
       s1.price AS cena_2022,
       s2.price AS cena_2023,
       (s2.price - s1.price) AS roznica
FROM products p1
JOIN sales s1 ON p1.product_id = s1.product_id
JOIN sales s2 ON p1.product_id = s2.product_id
WHERE EXTRACT(YEAR FROM s1.sale_date) = 2022
  AND EXTRACT(YEAR FROM s2.sale_date) = 2023
  AND TO_CHAR(s1.sale_date, 'MM-DD') = TO_CHAR(s2.sale_date, 'MM-DD');

8.
SELECT 
    p.product_category, 
    p.product_name, 
    s.price,
    MIN(s.price) OVER (PARTITION BY p.product_category) AS min_w_kat,
    MAX(s.price) OVER (PARTITION BY p.product_category) AS max_w_kat,
    (MAX(s.price) OVER (PARTITION BY p.product_category) - MIN(s.price) OVER (PARTITION BY p.product_category)) AS roznica_max_min
FROM sales s
JOIN products p ON s.product_id = p.product_id;

9.
SELECT 
    p.product_name, 
    s.sale_date,
    s.price,
    AVG(s.price) OVER (PARTITION BY p.product_id 
                       ORDER BY s.sale_date 
                       ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING) AS srednia_kroczaca
FROM sales s
JOIN products p ON s.product_id = p.product_id;

10.
SELECT 
    p.product_name, 
    p.product_category, 
    s.price,
    RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_cen,
    ROW_NUMBER() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS numer_wiersza,
    DENSE_RANK() OVER (PARTITION BY p.product_category ORDER BY s.price DESC) AS ranking_gesty
FROM sales s
JOIN products p ON s.product_id = p.product_id;

11.
SELECT 
    e.last_name, 
    p.product_name,
    s.sale_date,
    SUM(s.price * s.quantity) OVER (PARTITION BY e.employee_id ORDER BY s.sale_date) AS wartosc_rosnaca,
    RANK() OVER (ORDER BY s.price * s.quantity DESC) AS ranking_globalny_zamowienia
FROM sales s
JOIN employees e ON s.employee_id = e.employee_id
JOIN products p ON s.product_id = p.product_id;

12.
SELECT DISTINCT 
    e.first_name, 
    e.last_name, 
    e.job_id
FROM employees e
JOIN sales s ON e.employee_id = s.employee_id;
